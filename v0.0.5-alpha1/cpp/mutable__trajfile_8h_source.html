567  template <typename T, std::size_t k>
568  EigenVectorXkr_T<T, k> getVar(const nc::NcVar &var,
569  const std::size_t idx) const {
570  assert(idx < nFrames());
571 
572  nc_vlen_t vlenData;
573  var.getVar({idx}, &vlenData);
574 
575  // Initialize an Eigen object and copy the data over
576  EigenVectorXkr_T<T, k> vec(vlenData.len / k, k);
577  // Bind to nc_vlen_t memory and copy data over
578  vec = AlignedEigenMap_T<T, k, Eigen::RowMajor>(static_cast<T *>(vlenData.p),
579  vlenData.len / k, k);
580  return vec;
581  }

646  void initializeConventions() {
647 
648  const int compression_level = 5;
649 
650  // initialize data
651  fd->putAtt(CONVENTIONS_NAME, CONVENTIONS_VALUE);
652  fd->putAtt(CONVENTIONS_VERSION_NAME, CONVENTIONS_VERSION_VALUE);
653 
654  parameter_group = fd->addGroup(PARAM_GROUP_NAME);
655  traj_group = fd->addGroup(TRAJ_GROUP_NAME);
656 
657  frame_dim = traj_group.addDim(FRAME_NAME);
658 
659  time_var = traj_group.addVar(TIME_VAR, netCDF::ncDouble, {frame_dim});
660  time_var.putAtt(UNITS, TIME_UNITS);
661  time_var.setCompression(true, true, compression_level);
662 
663  uint_array_t = traj_group.addVlenType(UINT_ARR, nc::ncUint);
664  double_array_t = traj_group.addVlenType(DOUBLE_ARR, nc::ncDouble);
665 
666  topo_var = traj_group.addVar(TOPO_VAR, uint_array_t, {frame_dim});
667  topo_var.setCompression(true, true, compression_level);
668  coord_var = traj_group.addVar(COORD_VAR, double_array_t, {frame_dim});
669  coord_var.setCompression(true, true, compression_level);
670  phi_var = traj_group.addVar(PHI_VAR, double_array_t, {frame_dim});
671  phi_var.setCompression(true, true, compression_level);
672  vel_var = traj_group.addVar(VEL_VAR, double_array_t, {frame_dim});
673  vel_var.setCompression(true, true, compression_level);
