236  void createNewFile(const std::string &filename, gcs::SurfaceMesh &mesh,
237  gcs::VertexPositionGeometry &refVpg,
238  const NcFile::FileMode fMode) {
239  if (fd != nullptr) {
240  throw std::runtime_error("Cannot open an already open ...");
241  }
242 
243  writeable = true;
244 
245  fd = new NcFile(filename, fMode);
246  // initialize data
247  fd->putAtt(CONVENTIONS_NAME, CONVENTIONS_VALUE);
248  fd->putAtt(CONVENTIONS_VERSION_NAME, CONVENTIONS_VERSION_VALUE);
249 
250  frame_dim = fd->addDim(FRAME_NAME);
251  npolygons_dim = fd->addDim(NPOLYGONS_NAME, mesh.nFaces());
252  nvertices_dim = fd->addDim(NVERTICES_NAME, mesh.nVertices());
253  ncorners_dim = fd->addDim(NCORNERS_NAME, mesh.nCorners());
254  spatial_dim = fd->addDim(SPATIAL_DIMS_NAME, SPATIAL_DIMS);
255  polygon_order_dim = fd->addDim(POLYGON_ORDER_NAME, POLYGON_ORDER);
256 
257  // Initialize topology data block
258  topology =
259  fd->addVar(TOPO_VAR, nc::ncUint, {npolygons_dim, polygon_order_dim});
260 
261  // Populate topology data
262  Eigen::Matrix<std::uint32_t, Eigen::Dynamic, 3, Eigen::RowMajor>
263  faceMatrix = getFaceVertexMatrix(mesh);
264  std::uint32_t *topodata = faceMatrix.data();
265  topology.putVar(topodata);
266 
267  // Initialize reference coordinate data block
268  refcoord = fd->addVar(REFCOORD_VAR, netCDF::ncDouble,
269  {nvertices_dim, spatial_dim});
270  refcoord.putAtt(UNITS, LEN_UNITS);
271 
272  // Populate reference coordinate data
273  double *refcoorddata;
274  refcoorddata = gc::EigenMap<double, 3>(refVpg.inputVertexPositions).data();
275  refcoord.putVar(refcoorddata);

461  void getNcFrame(int &frame) const {
462  int maxFrame = getNextFrameIndex() - 1;
463  if (frame > maxFrame || frame < -(maxFrame + 1)) {
464  throw std::runtime_error("Snapshot frame exceed limiting frame index!");
465  } else if (frame < 0) {
466  frame = frame + maxFrame + 1;
467  }
468  }
